namespace = gai_ai_armies

event = {
	id = gai_ai_armies.1
	hide_window = yes
    is_triggered_only = yes 
	
	immediate = {
        every_country = {
            if = {
                limit = {
                    is_ai = yes
					NOT = { has_global_flag = disable_gai_armies } #AI will not build armies if this is disabled. You will need to enable army building in the defines
					count_armies = {
						limit = { army_type = assault_army }
						count < 15
					}
                    OR = {
						is_country_type = "default"
						is_country_type = awakened_fallen_empire
						
					}
					has_monthly_income = {
						resource = energy
						value > 30
					}
					energy > 500
					minerals > 100
                }
                country_event = { id = gai_ai_armies.2 }
            }
        }
	}
}


country_event = {
	id = gai_ai_armies.2	
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		random_owned_planet = {
			limit = {
				is_occupied_flag = no
				has_orbital_bombardment = no
			}
			while = {
				count = 26
				create_army = {
					owner = ROOT
					type = "assault_army"
				}
			}
		}
		add_minerals = -100
	}
}
