
namespace = ccrebel_petruxa


# Unrest: Greens
planet_event = {
	id = ccrebel_petruxa.1
	title = ccrebel_petruxa.1.name
	desc = ccrebel_petruxa.1.desc
	picture = GFX_evt_alien_nature
	show_sound = event_celebration
	location = root

	is_triggered_only = yes
	
	immediate = {
		add_modifier = {
			modifier = green_unrest
			days = 7200 # 20 years
		}
		set_timed_planet_flag = {
			flag = cc_revolt_imminent
			days = 7200
		}
		set_timed_planet_flag = {
			flag = recent_unrest_event
			days = 180
		}
	}

	option = {
		name = OK
		tooltip = {
			add_modifier = {
				modifier = green_unrest
				days = 7200 # 20 years
			}
		}
	}
}
