		text = CrisisManagerMenu.4810.desc4
		trigger = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarKhanMultiple value = 4 } } }
	}
	desc = {
		text = CrisisManagerMenu.4810.desc5
		trigger = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarKhanMultiple value = 5 } } }
	}
	desc = {
		text = CrisisManagerMenu.4810.desc
		trigger = {
			event_target:CmtGlobalVar = {
				NOR = {
					check_variable = { which = CmtVarKhanMultiple value = 1 }
					check_variable = { which = CmtVarKhanMultiple value = 2 }
					check_variable = { which = CmtVarKhanMultiple value = 3 }
					check_variable = { which = CmtVarKhanMultiple value = 4 }
					check_variable = { which = CmtVarKhanMultiple value = 5 }
				}
			}
		}
	}
	picture = GFX_evt_khan_throne_room
	is_triggered_only = yes
	
	after = {
		hidden_effect = {
			if = {
				limit = { has_country_flag = CmtFlagGoBack }
				remove_country_flag = CmtFlagGoBack
				country_event = { id = CrisisManagerMenu.4000 }
			} else_if = {
				limit = { has_country_flag = CmtFlagIndividual }
				country_event = { id = CrisisManagerMenu.1026 }
			} else_if = {
				limit = { event_target:CmtGlobalVar = { check_variable = { which = CmtVarKhanMultiple value = 1 } } }
				country_event = { id = CrisisManagerMenu.4000 }
			} else_if = {
				limit = {
					event_target:CmtGlobalVar = {
						OR = {
							check_variable = { which = CmtVarKhanMultiple value = 3 }
							check_variable = { which = CmtVarKhanMultiple value = 4 }
						}
					}
				}
				country_event = { id = CrisisManagerMenu.4820 }
			} else = { country_event = { id = CrisisManagerMenu.4830 } }
		}
	}
	
	option = {
		name = CrisisManagerKeepCurrent.name
		custom_tooltip = CrisisManagerKeepCurrent.tooltip
	}
	option = {
		name = CrisisManagerMenu.4810a.name
		custom_tooltip = CrisisManagerMenu.4810a.tooltip
