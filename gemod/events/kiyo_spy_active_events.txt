				days = 1800
			}
		}
		hidden_effect = {
			event_target:sabotageVictim = {
				random_system_within_border = {
					limit = {
						any_planet = {
							is_colony = yes
							owner = { is_country = event_target:sabotageVictim }
							OR = {
								is_capital = no
								owner = { num_owned_planets = 1 }
							}
						}
					}
					random_system_planet = {
						limit = { is_colony = no }
						save_event_target_as = bombA
					}
					random_system_planet = {
						limit = { is_colony = yes owner = { is_country = event_target:sabotageVictim } }
						save_event_target_as = bombB
					}
				}
				begin_event_chain = {
					event_chain = "kiyo_spy_active_bomb_chain"

				}

					name = "KIYO_SPY_ACTIVE_BOMB_A"

					limit = {
						NOR = {
							has_special_project = KIYO_SPY_ACTIVE_BOMB_A
							has_special_project = KIYO_SPY_ACTIVE_BOMB_B

			factor = 50
		}

			add_modifier = {
				modifier = kiyo_spy_active_bomb
				days = 1800
			}
		}

			event_target:sabotageVictim = {

							is_colony = yes

ship_event = {
	id = kiyo_event_active.321
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		solar_system = { save_event_target_as = kiyo_spy_active_bomb_system }
		if = {
			limit = {
				owner = {
					OR = {
						AND = {
							any_country = {
								has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_a }
								prev = { has_country_flag = kiyo_spy_active_bomb_victim_a }
							}
						}
						AND = {
							any_country = {
								has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_a }
								prev = { NOT = { has_country_flag = kiyo_spy_active_bomb_victim_a } }
							}
						}
					}
				}
			}
			if = {
				limit = {
					owner = {
						any_country = {
							has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_a }
							prev = { NOT = { has_country_flag = kiyo_spy_active_bomb_victim_a } }
						}
					}
				}
				owner = { add_modifier = { modifier = kiyo_spy_active_bomb_orbital days = 1800 } }
				event_target:kiyo_spy_active_bomb_system = {
					every_fleet_in_system = {
						limit = {
							any_owned_ship = {
								OR = {
									is_ship_class = shipclass_mining_station
									is_ship_class = shipclass_research_station
									is_ship_class = shipclass_observation_station
								}
							}
						}
						destroy_fleet = this
					}
				}
			}
			owner = { country_event = { id = kiyo_event_active.322 } }
		}
		else_if = {
			limit = {
				owner = {
					OR = {
						AND = {
							any_country = {
								has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_b }
								prev = { has_country_flag = kiyo_spy_active_bomb_victim_b }
							}
						}
						AND = {
							any_country = {
								has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_b }
								prev = { NOT = { has_country_flag = kiyo_spy_active_bomb_victim_b } }
							}
						}
					}
				}
			}
			if = {
				limit = {
					owner = {
						any_country = {
							has_relation_flag = { who = prev flag = kiyo_spy_active_bomb_b }
							prev = { NOT = { has_country_flag = kiyo_spy_active_bomb_victim_b } }
						}
					}
				}
				event_target:kiyo_spy_active_bomb_system = {
					every_system_planet = {
						limit = { is_colony = yes owner = { is_country_type = default } }
						kiyo_spy_blowup_planet = yes
					}
				}
			}
			owner = { country_event = { id = kiyo_event_active.323 } }
		}
	}
}
