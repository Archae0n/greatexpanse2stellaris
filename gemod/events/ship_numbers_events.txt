namespace = rebalance

# This fires when the game starts up
event = {
	id = rebalance.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = { is_country_type = default	}
			add_modifier = {
				modifier = "optimized_ships"
				days = -1
			}
			
		}
	}
}
