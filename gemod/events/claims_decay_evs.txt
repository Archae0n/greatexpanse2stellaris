		months = @claims_decay_time_xenophobe
	}

		country_event = {

		}
	}
}

country_event = {
	id = claims_decay.20
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = {
				NOT = { 
					is_rival = root
					root = { is_rival = prev }
				}
				OR = {
					is_loyal_to  = root
					is_cordial_to = root
					is_friendly_to = root
					is_in_federation_with = root
					has_association_status = root
					has_defensive_pact = root
					AND = {
						root = { 
							has_ascension_perk = ap_shared_destiny 
							is_overlord_to = prev
						}
					}
					root = { has_edict = diplomatic_grants }
				}
			}
		
			random_system_within_border = {
				limit = {
					root = {
						has_claim = prev
					}
				}
				remove_claims = { 
					who = root 
					num_of_claims = 5 
				} 
			}
		}
	}
}
