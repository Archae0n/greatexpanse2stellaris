kiyo_spy_blowup_planet = {
	set_variable = { which = kiyo_spy_num_buildings value = 0 }
	set_variable = { which = kiyo_spy_num_pops value = 0 }
	every_tile = {
		limit = {
			has_building = yes
		}
		prev = { change_variable = { which = kiyo_spy_num_buildings value = 1 } }
	}
	every_tile = {
		limit = {
			has_pop = yes
		}
		prev = { change_variable = { which = kiyo_spy_num_pops value = 1 } }
	}
	set_variable = { which = kiyo_spy_half_pops value = kiyo_spy_num_pops }
	divide_variable = { which = kiyo_spy_half_pops value = 2 }
	kiyo_spy_blowup_planet_looper = yes
}

kiyo_spy_blowup_planet_looper = {
	log = "Buildings [this.kiyo_spy_num_buildings]"
	log = "Pops [this.kiyo_spy_half_pops]"
	if = {
		limit = {
			check_variable = { which = kiyo_spy_num_buildings value > 0 }
			check_variable = { which = kiyo_spy_half_pops value > 0 }
		}
		if = {
			limit = {
				OR = {
					check_variable = { which = kiyo_spy_half_pops value > 0 }
					AND = {
						is_capital = yes
						num_pops > 8
					}
				}
			}
			random_tile = {
				limit = { has_building = yes }
				set_blocker = tb_bombarded_land
			}
		}
		subtract_variable = { which = kiyo_spy_num_buildings value = 1 }
		subtract_variable = { which = kiyo_spy_half_pops value = 1 }
	}
	if = {
		limit = {
			check_variable = { which = kiyo_spy_num_buildings value > 0 }
			check_variable = { which = kiyo_spy_half_pops value > 0 }
		}
		kiyo_spy_blowup_planet_looper = yes
	}
}

kiyo_spy_destroy_ships_looper = {
	if = {
		limit = { has_country_flag = kiyo_spy_destroyed_1 }
		subtract_variable = { which = fleetSizeDestroy value = 1 }
		remove_country_flag = kiyo_spy_destroyed_1
	}
	if = {
		limit = { has_country_flag = kiyo_spy_destroyed_2 }
		subtract_variable = { which = fleetSizeDestroy value = 2 }
		remove_country_flag = kiyo_spy_destroyed_2
	}
	if = {
		limit = { has_country_flag = kiyo_spy_destroyed_4 }
		subtract_variable = { which = fleetSizeDestroy value = 4 }
		remove_country_flag = kiyo_spy_destroyed_4
	}
	if = {
		limit = { has_country_flag = kiyo_spy_destroyed_8 }
		subtract_variable = { which = fleetSizeDestroy value = 8 }
		remove_country_flag = kiyo_spy_destroyed_8
	}
	if = {
		limit = {
			check_variable = { which = fleetSizeDestroy value > 0 }
			any_owned_ship = {
				OR = {
					is_ship_size = corvette
					is_ship_size = destroyer
					is_ship_size = cruiser
					is_ship_size = battleship
				}
			}
		}
		random_owned_ship = {
			limit = {
				OR = {
					is_ship_size = corvette
					is_ship_size = destroyer
					is_ship_size = cruiser
					is_ship_size = battleship
				}
			}
			if = {
				limit = { is_ship_size = corvette }
				root = { set_country_flag = kiyo_spy_destroyed_1 }
			}
			else_if = {
				limit = { is_ship_size = destroyer }
				root = { set_country_flag = kiyo_spy_destroyed_2 }
			}
			else_if = {
				limit = { is_ship_size = cruiser }
				root = { set_country_flag = kiyo_spy_destroyed_4 }
			}
			else_if = {
				limit = { is_ship_size = battleship }
				root = { set_country_flag = kiyo_spy_destroyed_8 }
			}
			destroy_ship = this
		}
	}
	else_if = {
		limit = {
			check_variable = { which = fleetSizeDestroy value > 0 }
			any_owned_ship = {
				is_ship_class = shipclass_military
				NOR = {
					is_ship_size = corvette
					is_ship_size = destroyer
					is_ship_size = cruiser
					is_ship_size = battleship
				}
			}
		}
		set_variable = { which = fleetSizeDestroy value = 0 }
	}
	if = {
		limit = { check_variable = { which = fleetSizeDestroy value > 0 } }
		kiyo_spy_destroy_ships_looper = yes
	}
}

	set_variable = {
		which = spy_total
		value = 1
	}
	if = { limit = { has_country_flag = kiyo_spy_primitive } change_variable = { which = spy_total value = 1 } }

	every_country = {
		limit = {
			reverse_has_relation_flag = { who = root flag = kiyo_spy_nospy }
			OR = {
				reverse_has_relation_flag = { who = root flag = kiyo_spy_spy1 }
				reverse_has_relation_flag = { who = root flag = kiyo_spy_spy2 }
				reverse_has_relation_flag = { who = root flag = kiyo_spy_spy3 }
				reverse_has_relation_flag = { who = root flag = kiyo_spy_spy4 }
			}
		}
		root = {
			if = {
				limit = { any_country = { reverse_has_relation_flag = { who = root flag = kiyo_spy_spy1 } } }
				remove_relation_flag = { who = event_target:spyTarget flag = kiyo_spy_spy1 }
			}
			else_if = {
				limit = { any_country = { reverse_has_relation_flag = { who = root flag = kiyo_spy_spy2 } } }
				remove_relation_flag = { who = event_target:spyTarget flag = kiyo_spy_spy2 }
			}
			else_if = {
				limit = { any_country = { reverse_has_relation_flag = { who = root flag = kiyo_spy_spy3 } } }
				remove_relation_flag = { who = event_target:spyTarget flag = kiyo_spy_spy3 }
			}
			else_if = {
				limit = { any_country = { reverse_has_relation_flag = { who = root flag = kiyo_spy_spy4 } } }
				remove_relation_flag = { who = event_target:spyTarget flag = kiyo_spy_spy4 }
			}
		}
	}
	kiyo_spy_calculate_spy = yes
}

		limit = {
			check_variable = { which = kiyo_spy_free_speech value > 0 }

		}

			limit = {
				is_species = prev

					pop_has_trait = trait_nerve_stapled
