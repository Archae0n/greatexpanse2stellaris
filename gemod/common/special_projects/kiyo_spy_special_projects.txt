	cost = 0
	days_to_research = 180
	timelimit = 500
	picture = GFX_evt_asteroid_field
	
	event_scope = ship_event

		leader = general
	}

		owner = { set_country_flag = kiyo_spy_active_bomb_victim_a }

	}

	}
