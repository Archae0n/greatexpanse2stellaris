building_power_hydroponics_farm_1 = {
	tags = { hydroponics_farm }

	base_buildtime = @b1time
	
	cost = {
		minerals = @b1cost
	}
	
	produced_resources = {
		food = 2
		unity = 1
		energy = 1
	}
	
	produced_resource_trigger = {
		modifier = {
			has_swapped_tradition = tr_adaptability_finish
			resources = {
				minerals = 1
			}
		}
	}
	
	upgrades = {
		building_power_hydroponics_farm_2
	}	

	prerequisites = {
		"tech_eco_simulation"
	}
	
	show_tech_unlock_if = { NOT = { has_authority = auth_machine_intelligence } }

	potential = {
		NOT = {
			owner = { has_authority = auth_machine_intelligence }
		}
		planet = {
			NOT = { is_planet_class = pc_habitat }	
		}
		owner = { has_valid_civic = civic_photoelectrolysis }
	}

	destroy_if = {
		exists = owner
		owner = { has_authority = auth_machine_intelligence }
	}
	
	ai_allow = {
		NOR = {
			has_resource = {
				type = sr_betharian
				amount > 0
			}	
			has_resource = {
				type = sr_alien_pets
				amount > 0
			}				
		}
	}

	ai_weight = {
		factor = 1
	}
}
