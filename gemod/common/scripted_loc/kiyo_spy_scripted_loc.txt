defined_text = {
	name = GetOverNavyCap
	text = {
		trigger = {
			used_naval_capacity_percent > 1
			check_variable = { which = intrigue value > 0 }
		}
		localization_key = kiyo_spy_overcap
	}
	text = {
		trigger = {
			used_naval_capacity_percent <= 1
		}
		localization_key = kiyo_spy_noinfo
	}
}

defined_text = {
	name = GetStarving
	text = {
		trigger = {
			AND = {
				food = 0
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
			check_variable = { which = intrigue value > 0 }
		}
		localization_key = kiyo_spy_starving
	}
	text = {
		trigger = {
			NAND = {
				food = 0
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}
		localization_key = kiyo_spy_noinfo
	}
}
